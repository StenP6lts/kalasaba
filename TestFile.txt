class Payment 
{
    public Currency $currency;
 
    public int $amt;
 
    public function __construct(
        Currency $currency,
        int $amt,
    ) {
        $this->currency = $currency;
        $this->amt= $amt;
    }
}

class Payment
{
    public function __construct(
        public Currency $currency,
        public int $amt,
    ) {}
